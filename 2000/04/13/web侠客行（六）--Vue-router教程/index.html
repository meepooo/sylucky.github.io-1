    const Foo = { template: '<div><h1>foo {{ this.$route.name }}</h1></div>' }
    const Bar = { template: '<div><h1>User {{ this.$route.name }}</h1></div>' }
    const User = { template: '<div><h1>this is a user template</h1></div>' }

    const router = new VueRouter({
      routes: [
          {
              path: '/',
              name: 'homepage',
              components: {
                  default: Foo,
                  a: Bar,
                  b: User
              }
          }
      ]
    })

    const app = new Vue({
      router: router,
      el: '#app',
    })

</script>
